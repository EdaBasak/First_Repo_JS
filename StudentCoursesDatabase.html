function addStudents(txt) {
    let msg = txt.length+" chars, ";
    let a = txt.split("\n");
    msg += a.length+" lines, ";
    for (let s of a) {
      let std = parseStudent(s);
      keys.push(std.id); vals.push(std);
    }
    report(msg + keys.length+" students");
}
